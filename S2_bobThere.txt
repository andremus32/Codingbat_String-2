public boolean bobThere(String str) {
  char[] arrFromStr = str.toCharArray();
  boolean check = false;
  if(str.length() > 2){
    for(int i = 0; i < str.length()-2; i++){
      if((str.charAt(i) == 'b' || str.charAt(i) == 'B') && (str.charAt(i + 2) == 'b' || str.charAt(i + 2) == 'B')){
        check = true;
      }
    }
  }
  return check;
}
