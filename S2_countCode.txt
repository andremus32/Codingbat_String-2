//String-2 > countCode
public int countCode(String str) {
  int codeCounter = 0;
  if(str.length() > 3){
    for(int i  = 0; i < str.length()-3; i++){
      if(str.toLowerCase().charAt(i) == 'c'){
        if(str.toLowerCase().charAt(i+1) == 'o' && str.toLowerCase().charAt(i+3) == 'e'){
          codeCounter++;
        }
      }
    }
  }
  return codeCounter;
}
