//String-2 > sameStarChar
public boolean sameStarChar(String str) {
  boolean ifNoTwoStars = true;
  if(str.length() > 2){
    for(int i = 1; i < str.length()-1; i++){
      if(str.charAt(i) == '*' && str.charAt(i+1) != str.charAt(i-1)){
        ifNoTwoStars  = false;
      }
    }
  }
  return ifNoTwoStars;
}
